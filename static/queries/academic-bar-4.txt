#gráfico de barras que mostra a produção do NeuroMat por ano
#defaultView:BarChart
SELECT ?year (COUNT(?item) AS ?num_articles)
WHERE {
  { ?item wdt:P361 wd:Q18477654; wdt:P577 ?date_. }
  UNION
  { SERVICE wdsubgraph:scholarly_articles { ?item wdt:P361 wd:Q18477654; wdt:P577 ?date_. } }
  
  BIND(STR(YEAR(?date_)) AS ?year)
}
GROUP BY ?year
ORDER BY ?year
